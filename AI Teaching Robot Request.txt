// Teaching Robot - Arduino Code
// Basic educational robot with LED feedback and buzzer

// Pin definitions
const int redLED = 8;
const int greenLED = 9;
const int blueLED = 10;
const int buzzer = 11;
const int button1 = 2;
const int button2 = 3;
const int button3 = 4;
const int button4 = 5;

// Variables to store button states
int button1State = 0;
int button2State = 0;
int button3State = 0;
int button4State = 0;

// Quiz variables
int currentQuestion = 0;
int score = 0;
int totalQuestions = 3;

// Structure to hold questions (in a real robot, you might use an SD card for more storage)
struct Question {
  String question;
  String optionA;
  String optionB;
  String optionC;
  String optionD;
  char correctAnswer; // A, B, C, or D
};

// Sample questions
Question questions[] = {
  {"What is 5 + 3?", "7", "8", "9", "10", 'B'},
  {"How many legs does a spider have?", "6", "10", "8", "12", 'C'},
  {"What is the capital of France?", "London", "Berlin", "Paris", "Rome", 'C'}
};

void setup() {
  // Initialize serial communication
  Serial.begin(9600);
 
  // Initialize LED pins as outputs
  pinMode(redLED, OUTPUT);
  pinMode(greenLED, OUTPUT);
  pinMode(blueLED, OUTPUT);
  pinMode(buzzer, OUTPUT);
 
  // Initialize button pins as inputs
  pinMode(button1, INPUT);
  pinMode(button2, INPUT);
  pinMode(button3, INPUT);
  pinMode(button4, INPUT);
 
  // Welcome message
  Serial.println("Welcome to the Learning Robot!");
  Serial.println("Press buttons to answer questions.");
  Serial.println();
 
  // Initial LED sequence
  startupSequence();
 
  // Start first question
  askQuestion(currentQuestion);
}

void loop() {
  // Read button states
  button1State = digitalRead(button1);
  button2State = digitalRead(button2);
  button3State = digitalRead(button3);
  button4State = digitalRead(button4);
 
  // Check if any button is pressed
  if (button1State == HIGH || button2State == HIGH ||
      button3State == HIGH || button4State == HIGH) {
   
    // Small delay to debounce
    delay(300);
   
    // Determine which answer was selected
    char selectedAnswer;
    if (button1State == HIGH) selectedAnswer = 'A';
    else if (button2State == HIGH) selectedAnswer = 'B';
    else if (button3State == HIGH) selectedAnswer = 'C';
    else if (button4State == HIGH) selectedAnswer = 'D';
   
    // Check answer
    checkAnswer(selectedAnswer);
   
    // Move to next question or end quiz
    currentQuestion++;
    if (currentQuestion < totalQuestions) {
      delay(2000);
      askQuestion(currentQuestion);
    } else {
      endQuiz();
    }
  }
}

void startupSequence() {
  // Blue LED for startup
  digitalWrite(blueLED, HIGH);
  tone(buzzer, 523, 200); // C note
  delay(500);
  digitalWrite(blueLED, LOW);
  delay(500);
 
  // Green LED
  digitalWrite(greenLED, HIGH);
  tone(buzzer, 659, 200); // E note
  delay(500);
  digitalWrite(greenLED, LOW);
  delay(500);
 
  // Red LED
  digitalWrite(redLED, HIGH);
  tone(buzzer, 784, 200); // G note
  delay(500);
  digitalWrite(redLED, LOW);
  delay(500);
 
  // All LEDs together
  digitalWrite(redLED, HIGH);
  digitalWrite(greenLED, HIGH);
  digitalWrite(blueLED, HIGH);
  tone(buzzer, 1047, 500); // C note (octave higher)
  delay(1000);
  digitalWrite(redLED, LOW);
  digitalWrite(greenLED, LOW);
  digitalWrite(blueLED, LOW);
  noTone(buzzer);
}

void askQuestion(int qNumber) {
  // Clear the serial monitor
  Serial.println("\n---");
 
  // Ask the question
  Serial.println(questions[qNumber].question);
  Serial.println("A: " + questions[qNumber].optionA);
  Serial.println("B: " + questions[qNumber].optionB);
  Serial.println("C: " + questions[qNumber].optionC);
  Serial.println("D: " + questions[qNumber].optionD);
  Serial.println("Press the corresponding button (1-4) to answer:");
 
  // LED indication that question is ready
  digitalWrite(blueLED, HIGH);
}

void checkAnswer(char answer) {
  digitalWrite(blueLED, LOW);
 
  if (answer == questions[currentQuestion].correctAnswer) {
    // Correct answer
    Serial.println("Correct! Well done!");
    digitalWrite(greenLED, HIGH);
    tone(buzzer, 1047, 200); // High C
    delay(100);
    tone(buzzer, 1319, 400); // E
    delay(500);
    digitalWrite(greenLED, LOW);
    noTone(buzzer);
    score++;
  } else {
    // Incorrect answer
    Serial.println("Incorrect. Try again!");
    digitalWrite(redLED, HIGH);
    tone(buzzer, 200, 500); // Low frequency buzz
    delay(1000);
    digitalWrite(redLED, LOW);
    noTone(buzzer);
  }
 
  Serial.println("Correct answer was: " + String(questions[currentQuestion].correctAnswer));
}

void endQuiz() {
  Serial.println("\n=== QUIZ COMPLETE ===");
  Serial.println("Your final score: " + String(score) + " out of " + String(totalQuestions));
 
  // Play ending melody based on score
  if (score == totalQuestions) {
    // Perfect score melody
    tone(buzzer, 523, 200);
    delay(250);
    tone(buzzer, 659, 200);
    delay(250);
    tone(buzzer, 784, 200);
    delay(250);
    tone(buzzer, 1047, 600);
  } else if (score >= totalQuestions/2) {
    // Passing score melody
    tone(buzzer, 523, 200);
    delay(250);
    tone(buzzer, 659, 600);
  } else {
    // Try again melody
    tone(buzzer, 440, 400);
    delay(500);
    tone(buzzer, 349, 400);
  }
 
  // LED indication based on score
  if (score == totalQuestions) {
    // All LEDs for perfect score
    digitalWrite(redLED, HIGH);
    digitalWrite(greenLED, HIGH);
    digitalWrite(blueLED, HIGH);
    delay(2000);
    digitalWrite(redLED, LOW);
    digitalWrite(greenLED, LOW);
    digitalWrite(blueLED, LOW);
  } else {
    // Just green for passing, red for failing
    if (score >= totalQuestions/2) {
      digitalWrite(greenLED, HIGH);
    } else {
      digitalWrite(redLED, HIGH);
    }
    delay(2000);
    digitalWrite(greenLED, LOW);
    digitalWrite(redLED, LOW);
  }
 
  noTone(buzzer);
 
  // Wait before restarting
  delay(3000);
  Serial.println("Restarting quiz...");
  currentQuestion = 0;
  score = 0;
  askQuestion(currentQuestion);
}

    def math_quiz(self):
        print("\n🧮 MATH QUIZ ACTIVITY")
        print("-" * 30)
       
        # Adjust range based on difficulty
        ranges = {1: (1, 10), 2: (10, 50), 3: (20, 100)}
        min_val, max_val = ranges[self.difficulty]
       
        operations = ['+', '-', '*', '/']
        if self.difficulty == 1:
            operations = ['+', '-']
        elif self.difficulty == 2:
            operations = ['+', '-', '*']
           
        correct = 0
        total = 5 if self.difficulty == 1 else (7 if self.difficulty == 2 else 10)
       
        for i in range(total):
            num1 = random.randint(min_val, max_val)
            num2 = random.randint(min_val, max_val)
           
            if self.difficulty == 3 and i % 2 == 0:
                # For hard level, include some division
                if num2 != 0 and num1 % num2 == 0:
                    operation = '/'
                else:
                    operation = random.choice(['+', '-', '*'])
            else:
                operation = random.choice(operations)
               
            # Ensure positive results for subtraction
            if operation == '-' and num1 < num2:
                num1, num2 = num2, num1
               
            if operation == '+':
                answer = num1 + num2
            elif operation == '-':
                answer = num1 - num2
            elif operation == '*':
                answer = num1 * num2
            elif operation == '/' and num2 != 0:
                answer = num1 // num2
                num1 = answer * num2  # Ensure clean division
               
            question = f"{num1} {operation} {num2} = "
           
            try:
                user_answer = int(input(f"Q{i+1}: {question}"))
                if user_answer == answer:
                    print("✅ Correct!")
                    correct += 1
                    self.score += self.difficulty
                else:
                    print(f"❌ Wrong! The answer is {answer}")
            except ValueError:
                print("Please enter a number!")
                print(f"The answer is {answer}")
               
            self.total_questions += 1
            time.sleep(0.5)
           
        print(f"\n📊 Math Score: {correct}/{total}")

    int getDifficulty() {
        int level;
        while (true) {
            std::cout << "\nChoose difficulty (1-Easy, 2-Medium, 3-Hard): ";
            std::cin >> level;
           
            if (level >= 1 && level <= 3) {
                difficulty = level;
                return level;
            } else {
                std::cout << "Please enter 1, 2, or 3" << std::endl;
            }
        }
    }
   
    int generateRandom(int min, int max) {
        std::uniform_int_distribution<> dis(min, max);
        return dis(gen);
    }
   
    void clearInputBuffer() {
        std::cin.clear();
        std::cin.ignore(10000, '\n');
    }
   
    std::pair<int, int> mathQuiz() {
        std::cout << "\n🧮 MATH QUIZ ACTIVITY" << std::endl;
        std::cout << std::string(30, '-') << std::endl;
       
        // Adjust range based on difficulty
        int min_val, max_val;
        if (difficulty == 1) {
            min_val = 1; max_val = 10;
        } else if (difficulty == 2) {
            min_val = 10; max_val = 50;
        } else {
            min_val = 20; max_val = 100;
        }
       
        std::vector<char> operations = {'+', '-', '*'};
        if (difficulty == 1) {
            operations = {'+', '-'};
        } else if (difficulty == 2) {
            operations = {'+', '-', '*'};
        }
       
        int correct = 0;
        int total = (difficulty == 1) ? 5 : ((difficulty == 2) ? 7 : 10);
       
        for (int i = 0; i < total; i++) {
            int num1 = generateRandom(min_val, max_val);
            int num2 = generateRandom(min_val, max_val);
           
            char operation = operations[generateRandom(0, operations.size() - 1)];
           
            // Ensure positive results for subtraction
            if (operation == '-' && num1 < num2) {
                std::swap(num1, num2);
            }
           
            int answer;
            if (operation == '+') {
                answer = num1 + num2;
            } else if (operation == '-') {
                answer = num1 - num2;
            } else if (operation == '*') {
                answer = num1 * num2;
            }
           
            std::cout << "Q" << (i+1) << ": " << num1 << " " << operation << " " << num2 << " = ";
           
            int userAnswer;
            std::cin >> userAnswer;
           
            if (userAnswer == answer) {
                std::cout << "✅ Correct!" << std::endl;
                correct++;
                score += difficulty;
            } else {
                std::cout << "❌ Wrong! The answer is " << answer << std::endl;
            }
           
            totalQuestions++;
            std::this_thread::sleep_for(std::chrono::milliseconds(500));
        }
       
        std::cout << "\n📊 Math Score: " << correct << "/" << total << std::endl;
        return std::make_pair(correct, total);
    }
   
    std::pair<int, int> vocabularyBuilder() {
        std::cout << "\n📖 VOCABULARY BUILDER" << std::endl;
        std::cout << std::string(30, '-') << std::endl;
       
        // Vocabulary database by difficulty
        std::map<std::string, std::string> vocabEasy = {
            {"happy", "feeling joy or pleasure"},
            {"big", "of considerable size"},
            {"small", "not large in size"},
            {"fast", "moving quickly"},
            {"slow", "moving at a low speed"}
        };
       
        std::map<std::string, std::string> vocabMedium = {
            {"enormous", "very large in size"},
            {"mysterious", "full of mystery; difficult to understand"},
            {"furious", "extremely angry"},
            {"delicious", "highly pleasant to the taste"},
            {"ancient", "belonging to the very distant past"}
        };
       
        std::map<std::string, std::string> vocabHard = {
            {"ephemeral", "lasting for a very short time"},
            {"ubiquitous", "present, appearing, or found everywhere"},
            {"serendipity", "the occurrence of events by chance in a happy way"},
            {"eloquent", "fluent or persuasive in speaking or writing"},
            {"meticulous", "showing great attention to detail"}
        };
       
        std::map<std::string, std::string> currentVocab;
        if (difficulty == 1) currentVocab = vocabEasy;
        else if (difficulty == 2) currentVocab = vocabMedium;
        else currentVocab = vocabHard;
       
        // Convert map to vector for shuffling
        std::vector<std::pair<std::string, std::string>> vocabList;
        for (const auto& pair : currentVocab) {
            vocabList.push_back(pair);
        }
       
        std::shuffle(vocabList.begin(), vocabList.end(), gen);
       
        int correct = 0;
        int total = (difficulty == 1) ? 3 : ((difficulty == 2) ? 4 : 5);
       
        for (int i = 0; i < std::min(total, (int)vocabList.size()); i++) {
            std::string word = vocabList[i].first;
            std::string definition = vocabList[i].second;
           
            std::cout << "\nQ" << (i+1) << ": What word means '" << definition << "'?" << std::endl;
           
            // Create options
            std::vector<std::string> options = {word};
            std::vector<std::string> allWords;
            for (const auto& pair : currentVocab) {
                if (pair.first != word) {
                    allWords.push_back(pair.first);
                }
            }
           
            std::shuffle(allWords.begin(), allWords.end(), gen);
            for (int j = 0; j < std::min(3, (int)allWords.size()) && options.size() < 4; j++) {
                options.push_back(allWords[j]);
            }
           
            std::shuffle(options.begin(), options.end(), gen);
           
            for (int j = 0; j < options.size(); j++) {
                std::cout << "   " << char('A' + j) << ". " << options[j] << std::endl;
            }
           
            std::cout << "Your answer (A/B/C/D): ";
            char answer;
            std::cin >> answer;
            answer = std::toupper(answer);
           
            if (answer >= 'A' && answer < 'A' + options.size()) {
                int index = answer - 'A';
                if (options[index] == word) {
                    std::cout << "✅ Correct!" << std::endl;
                    correct++;
                    score += difficulty;
                } else {
                    std::cout << "❌ Wrong! The correct answer is '" << word << "'" << std::endl;
                }
            } else {
                std::cout << "Invalid choice! The correct answer is '" << word << "'" << std::endl;
            }
           
            totalQuestions++;
            std::this_thread::sleep_for(std::chrono::milliseconds(500));
        }
       
        std::cout << "\n📊 Vocabulary Score: " << correct << "/" << total << std::endl;
        return std::make_pair(correct, total);
    }

    std::pair<int, int> wordScramble() {
        std::cout << "\n🔤 WORD SCRAMBLE" << std::endl;
        std::cout << std::string(30, '-') << std::endl;
       
        std::vector<std::string> wordsEasy = {"cat", "dog", "sun", "hat", "pen"};
        std::vector<std::string> wordsMedium = {"water", "house", "apple", "grape", "music"};
        std::vector<std::string> wordsHard = {"computer", "elephant", "hospital", "astronaut", "chocolate"};
       
        std::vector<std::string> currentWords;
        if (difficulty == 1) currentWords = wordsEasy;
        else if (difficulty == 2) currentWords = wordsMedium;
        else currentWords = wordsHard;
       
        std::shuffle(currentWords.begin(), currentWords.end(), gen);
       
        int correct = 0;
        int total = (difficulty == 1) ? 3 : ((difficulty == 2) ? 4 : 5);
       
        for (int i = 0; i < std::min(total, (int)currentWords.size()); i++) {
            std::string word = currentWords[i];
           
            // Scramble the word
            std::string scrambled = word;
            std::shuffle(scrambled.begin(), scrambled.end(), gen);
           
            // Make sure it's different from original
            if (scrambled == word) {
                std::shuffle(scrambled.begin(), scrambled.end(), gen);
            }
           
            std::cout << "Q" << (i+1) << ": Unscramble '" << scrambled << "': ";
           
            std::string userAnswer;
            std::cin >> userAnswer;
           
            if (userAnswer == word) {
                std::cout << "✅ Correct!" << std::endl;
                correct++;
                score += difficulty;
            } else {
                std::cout << "❌ Wrong! The word is '" << word << "'" << std::endl;
            }
           
            totalQuestions++;
            std::this_thread::sleep_for(std::chrono::milliseconds(500));
        }
       
        std::cout << "\n📊 Word Scramble Score: " << correct << "/" << total << std::endl;
        return std::make_pair(correct, total);
    }
   
    void showProgress() {
        if (totalQuestions > 0) {
            double accuracy = (double(score) / (totalQuestions * difficulty)) * 100;
            std::cout << "\n📊 YOUR PROGRESS:" << std::endl;
            std::cout << "   Total Questions: " << totalQuestions << std::endl;
            std::cout << "   Score: " << score << std::endl;
            std::cout << "   Accuracy: " << std::fixed << std::setprecision(1) << accuracy << "%" << std::endl;
           
            // Encouragement based on performance
            if (accuracy >= 90) {
                std::cout << "   🌟 Excellent work! Keep it up!" << std::endl;
            } else if (accuracy >= 70) {
                std::cout << "   👍 Good job! Keep practicing!" << std::endl;
            } else {
                std::cout << "   💪 You're making progress! Don't give up!" << std::endl;
            }
        } else {
            std::cout << "No activities completed yet!" << std::endl;
        }
    }
   
    void runActivity(int choice) {
        switch(choice) {
            case 1:
                getDifficulty();
                mathQuiz();
                break;
            case 2:
                getDifficulty();
                vocabularyBuilder();
                break;
            case 3:
                getDifficulty();
                scienceTrivia();
                break;
            case 5:
                getDifficulty();
                wordScramble();
                break;
            default:
                std::cout << "Activity not yet implemented in C++ version." << std::endl;
                break;
        }
    }
   
    void run() {
        greetStudent();
       
        while (true) {
            showActivities();
            std::cout << "\nSelect an activity (0-6): ";
           
            int choice;
            std::cin >> choice;
           
            if (choice == 0) {
                std::cout << "\n👋 Goodbye, " << studentName << "! Thanks for learning with me!" << std::endl;
                showProgress();
                break;
            } else if (choice >= 1 && choice <= 6) {
                runActivity(choice);
                std::cout << "\n" << std::string(50, '=') << std::endl;
            } else {
                std::cout << "Please select a valid option!" << std::endl;
            }
        }
    }
